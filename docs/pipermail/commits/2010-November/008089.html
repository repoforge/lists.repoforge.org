+%clean

+%post

+%preun

+    export GCONF_CONFIG_SOURCE="$(gconftool-2 --get-default-source)"
+    gconftool-2 --makefile-uninstall-rule %{_sysconfdir}/gconf/schemas/%{name}.schemas > /dev/null || :
