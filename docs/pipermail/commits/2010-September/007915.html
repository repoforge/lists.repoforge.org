+++ trunk/rpms/tidy/tidy-cvs_checkout.sh Thu Sep 23 17:55:30 2010
@@ -1,0 +1,22 @@
+#!/bin/bash
+
+if [ $# -gt 0 ]; then  
+  DATE=$1
+  shift
+else
+  DATE=$(date +%Y%m%d)
+fi
+
+MODULE="$(basename $0 -cvs_checkout.sh)"
+
+set -x
+rm -rf $MODULE
+
+cvs -z3 -d:pserver:anonymous at tidy.cvs.sourceforge.net:/cvsroot/tidy export -D $DATE $MODULE 
+
+## tar it up
+tar czf $MODULE-${DATE}cvs.tar.gz $MODULE
+
+## cleanup
+rm -rf $MODULE
+
