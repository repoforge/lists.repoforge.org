+#!/bin/bash
+#
+# Init file for Jungledisk backup daemon
+#
+# Written by Dag Wieers <dag at wieers.com>
+#
+# chkconfig: - 54 46
+# description: Jungledisk backup
+#
+# processname: jungledisk
+# config: %{_sysconfdir}/jungledisk-settings.ini
+# pidfile: %{_localstatedir}/run/jungledisk
+
+source %{_initrddir}/functions
+
+[ -x %{_bindir}/jungledisk ] || exit 1
+[ -r %{_sysconfdir}/jungledisk-settings.ini ] || exit 1
+
+RETVAL=0
+prog="jungledisk"
+desc="Jungledisk daemon"
+
+start() {
+    echo -n $"Starting $desc ($prog): "
+    daemon $prog -c %{_sysconfdir}/jungledisk-settings.ini
+    RETVAL=$?
+    echo
+    [ $RETVAL -eq 0 ] && touch %{_localstatedir}/lock/subsys/$prog
+    return $RETVAL
+}
+
+stop() {
+    echo -n $"Shutting down $desc ($prog): "
+    killproc $prog
+    RETVAL=$?
+    echo
+    [ $RETVAL -eq 0 ] && rm -f %{_localstatedir}/lock/subsys/$prog
+    return $RETVAL
+}
+
+restart() {
+    stop
+    start
+}
+
+reload() {
+    echo -n $"Reloading $desc ($prog): "
+    killproc $prog -HUP
+    RETVAL=$?
+    echo
+    return $RETVAL
+}
+
+case "$1" in
+  start)
+    start
+    ;;
+  stop)
+    stop
+    ;;
+  restart)
+    restart
+    ;;
+  reload)
+    reload
+    ;;
+  condrestart)
+    [ -e %{_localstatedir}/lock/subsys/$prog ] && restart
+    RETVAL=$?
+    ;;
+  status)
+    status $prog
+    RETVAL=$?
+    ;;
+  *)
+    echo $"Usage: $0 {start|stop|restart|reload|condrestart|status}"
+    RETVAL=1
+esac
+
+exit $RETVAL
