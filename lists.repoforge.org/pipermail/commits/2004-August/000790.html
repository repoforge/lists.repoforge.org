+Short explanation of directory structure
+""""""""""""""""""""""""""""""""""""""""
+
+srcdir = /var/yam
+
+	/extras		-	- Yam managed (rsync/ftp/http)
+	/iso		-	- Manually managed (add your dist ISOs here)
+	/local		-	- Manually managed (add your 'own' packages here per dist-arch)
+		eg.
+
+		/all		- Inserted in all distributions (generic packages)
+		/el3-x86_64	- Enterprise Linux 3 / x86_64
+		/fc2-i386	- Fedora Core 2 / i386
+
+	/os		-	- Yam managed (rsync/ftp/http)
+	/updates	-	- Yam manager (rsync/ftp/http)
+
+wwwdir = /var/www/yam
+
+	eg.
+		/el3-x86_64	- Enterprise Linux 3 / x86_64
+		/fc2-i386	- Fedora Core 2 / i386
+
+	per dist-arch one directory, autogenerated
+	you can remove everything from here and regenerated with Yam
+	please don't add packages here, see /var/yam/local !

+Naming convention for extra packages
+""""""""""""""""""""""""""""""""""""
+
+	<name>-<version>-<release>.<disttag>.<repotag>.<arch>.rpm
+
+		where
+			name		usually lowercase package-name
+			version		version containing only numerics
+			release		release may contain rc, beta, alpha, cvs
+			disttag		rh6, rh7, rh8, rh9, el2, el3, fc1, fc2, sles8, ...
+					generic packageis (dist-independent) have 0 as disttag
+			repotag		dag, fr, dries 
+			arch		noarch, i386, i486, i586, i686, x86_64, ppc, s390, ...
